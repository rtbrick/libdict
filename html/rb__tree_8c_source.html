148size_t
149rb_tree_clear(rb_tree* tree, dict_delete_func delete_func)
150{
151 const size_t count = tree->count;
152
153 rb_node* node = tree->root;
154 while (node) {
155 if (node->llink || node->rlink) {
156 node = node->llink ? node->llink : node->rlink;
157 continue;
158 }
159
160 if (delete_func)
161 delete_func(node->key, node->datum);
162
163 rb_node* const parent = PARENT(node);
164 FREE(node);
165 *(parent ? (parent->llink == node ? &parent->llink : &parent->rlink) : &tree->root) = NULL;
166 node = parent;
167 }
168 ASSERT(tree->root == NULL);
169 tree->count = 0;
170 return count;
171}

179dict_insert_result
180rb_tree_insert(rb_tree* tree, void* key)
181{
182 int cmp = 0; /* Quell GCC warning about uninitialized usage. */
183 rb_node* node = tree->root;
184 rb_node* parent = NULL;
185 while (node) {
186 cmp = tree->cmp_func(key, node->key);
187 if (cmp < 0) {
188 parent = node; node = node->llink;
189 } else if (cmp > 0) {
190 parent = node; node = node->rlink;
191 } else
192 return (dict_insert_result) { &node->datum, false };
193 }
194
195 if (!(node = node_new(key)))
196 return (dict_insert_result) { NULL, false };
197
198 SET_PARENT(node, parent);
199 if (parent == NULL) {
200 ASSERT(tree->root == NULL);
201 tree->root = node;
202 ASSERT(tree->count == 0);
203 SET_BLACK(node);
204 } else {
205 if (cmp < 0)
206 parent->llink = node;
207 else
208 parent->rlink = node;
209
210 tree->rotation_count += insert_fixup(tree, node);
211 }
212 tree->count++;
213 return (dict_insert_result) { &node->datum, true };
214}

216static unsigned
217insert_fixup(rb_tree* tree, rb_node* node)
218{
219 unsigned rotations = 0;
220 while (node != tree->root && COLOR(PARENT(node)) == RB_RED) {
221 if (PARENT(node) == PARENT(PARENT(node))->llink) {
222 rb_node* temp = PARENT(PARENT(node))->rlink;
223 if (temp && COLOR(temp) == RB_RED) {
224 SET_BLACK(temp);
225 node = PARENT(node);
226 SET_BLACK(node);
227 node = PARENT(node);
228 SET_RED(node);
229 } else {
230 if (node == PARENT(node)->rlink) {
231 node = PARENT(node);
232 rot_left(tree, node);
233 ++rotations;
234 }
235 temp = PARENT(node);
236 SET_BLACK(temp);
237 temp = PARENT(temp);
238 SET_RED(temp);
239 rot_right(tree, temp);
240 ++rotations;
241 }
242 } else {
243 rb_node* temp = PARENT(PARENT(node))->llink;
244 if (temp && COLOR(temp) == RB_RED) {
245 SET_BLACK(temp);
246 node = PARENT(node);
247 SET_BLACK(node);
248 node = PARENT(node);
249 SET_RED(node);
250 } else {
251 if (node == PARENT(node)->llink) {
252 node = PARENT(node);
253 rot_right(tree, node);
254 ++rotations;
255 }
256 temp = PARENT(node);
257 SET_BLACK(temp);
258 temp = PARENT(temp);
259 SET_RED(temp);
260 rot_left(tree, temp);
261 ++rotations;
262 }
263 }
264 }
265
266 SET_BLACK(tree->root);
267 return rotations;
268}

270static void
271remove_node(rb_tree* tree, rb_node* node)
272{
273 rb_node* out;
274 if (!node->llink || !node->rlink) {
275 out = node;
276 } else {
277 out = tree_node_min(node->rlink);
278 void *tmp;
279 SWAP(node->key, out->key, tmp);
280 SWAP(node->datum, out->datum, tmp);
281 }
282
283 rb_node* const x = out->llink ? out->llink : out->rlink;
284 rb_node* const xp = PARENT(out);
285 if (x)
286 SET_PARENT(x, xp);
287 bool left = xp && xp->llink == out;
288 *(xp ? (xp->llink == out ? &xp->llink : &xp->rlink) : &tree->root) = x;
289
290 if (COLOR(out) == RB_BLACK && tree->root)
291 tree->rotation_count += delete_fixup(tree, x, xp, left);
292 FREE(out);
293 tree->count--;
294}

307static unsigned
308delete_fixup(rb_tree* tree, rb_node* node, rb_node* parent, bool left)
309{
310 unsigned rotations = 0;
311 while (node != tree->root && (node == NULL || COLOR(node) == RB_BLACK)) {
312 if (left) {
313 rb_node* w = parent->rlink;
314 if (COLOR(w) == RB_RED) {
315 SET_BLACK(w);
316 SET_RED(parent);
317 rot_left(tree, parent); ++rotations;
318 w = parent->rlink;
319 }
320 if ((w->llink == NULL || COLOR(w->llink) == RB_BLACK) &&
321 (w->rlink == NULL || COLOR(w->rlink) == RB_BLACK)) {
322 SET_RED(w);
323 node = parent;
324 parent = PARENT(parent);
325 left = parent && parent->llink == node;
326 } else {
327 if (w->rlink == NULL || COLOR(w->rlink) == RB_BLACK) {
328 SET_BLACK(w->llink);
329 SET_RED(w);
330 rot_right(tree, w); ++rotations;
331 w = parent->rlink;
332 }
333 if (COLOR(parent) == RB_RED)
334 SET_RED(w);
335 else
336 SET_BLACK(w);
337 if (w->rlink)
338 SET_BLACK(w->rlink);
339 SET_BLACK(parent);
340 rot_left(tree, parent); ++rotations;
341 break;
342 }
343 } else { // left == false
344 rb_node* w = parent->llink;
345 if (COLOR(w) == RB_RED) {
346 SET_BLACK(w);
347 SET_RED(parent);
348 rot_right(tree, parent); ++rotations;
349 w = parent->llink;
350 }
351 if ((w->llink == NULL || COLOR(w->llink) == RB_BLACK) &&
352 (w->rlink == NULL || COLOR(w->rlink) == RB_BLACK)) {
353 SET_RED(w);
354 node = parent;
355 parent = PARENT(parent);
356 left = parent && parent->llink == node;
357 } else {
358 if (w->llink == NULL || COLOR(w->llink) == RB_BLACK) {
359 SET_BLACK(w->rlink);
360 SET_RED(w);
361 rot_left(tree, w); ++rotations;
362 w = parent->llink;
363 }
364 if (COLOR(parent) == RB_RED)
365 SET_RED(w);
366 else
367 SET_BLACK(w);
368 if (w->llink)
369 SET_BLACK(w->llink);
370 SET_BLACK(parent);
371 rot_right(tree, parent); ++rotations;
372 break;
373 }
374 }
375 }
376
377 if (node)
378 SET_BLACK(node);
379 return rotations;
380}

387size_t
388rb_tree_traverse(rb_tree* tree, dict_visit_func visit, void* user_data)
389{
390 if (tree->root == NULL)
391 return 0;
392
393 size_t count = 0;
394 rb_node* node = tree_node_min(tree->root);
395 for (; node != NULL; node = node_next(node)) {
396 ++count;
397 if (!visit(node->key, node->datum, user_data))
398 break;
399 }
400 return count;
401}

403bool
404rb_tree_select(rb_tree *tree, size_t n, const void **key, void **datum)
405{
406 if (n >= tree->count) {
407 if (key)
408 *key = NULL;
409 if (datum)
410 *datum = NULL;
411 return false;
412 }
413 rb_node *node;
414 if (n >= tree->count / 2) {
415 node = tree_node_max(tree->root);
416 n = tree->count - 1 - n;
417 while (n--)
418 node = node_prev(node);
419 } else {
420 node = tree_node_min(tree->root);
421 while (n--)
422 node = node_next(node);
423 }
424 if (key)
425 *key = node->key;
426 if (datum)
427 *datum = node->datum;
428 return true;
429}

431static void
432rot_left(rb_tree* tree, rb_node* node)
433{
434 rb_node* const rlink = node->rlink;
435 if ((node->rlink = rlink->llink) != NULL)
436 SET_PARENT(rlink->llink, node);
437 rb_node* const parent = PARENT(node);
438 SET_PARENT(rlink, parent);
439 *(parent ? (parent->llink == node ? &parent->llink : &parent->rlink) : &tree->root) = rlink;
440 rlink->llink = node;
441 SET_PARENT(node, rlink);
442}

453 llink->rlink = node;
454 SET_PARENT(node, llink);

508static bool
509node_verify(const rb_tree* tree, const rb_node* parent, const rb_node* node,
510 unsigned black_node_count, unsigned leaf_black_node_count)
511{
512 if (parent == NULL) {
513 VERIFY(tree->root == node);
514 VERIFY(node == NULL || COLOR(node) == RB_BLACK);
515 } else {
516 VERIFY(parent->llink == node || parent->rlink == node);
517 }
518 if (node) {
519 VERIFY(PARENT(node) == parent);
520 if (parent) {
521 if (parent->llink == node) {
522 VERIFY(tree->cmp_func(parent->key, node->key) > 0);
523 } else {
524 ASSERT(parent->rlink == node);
525 VERIFY(tree->cmp_func(parent->key, node->key) < 0);
526 }
527 }
528 if (COLOR(node) == RB_RED) {
529 /* Verify that every child of a red node is black. */
530 if (node->llink)
531 VERIFY(COLOR(node->llink) == RB_BLACK);
532 if (node->rlink)
533 VERIFY(COLOR(node->rlink) == RB_BLACK);
534 } else {
535 black_node_count++;
536 }
537 if (!node->llink && !node->rlink) {
538 /* Verify that each path to a leaf contains the same number of black nodes. */
539 VERIFY(black_node_count == leaf_black_node_count);
540 }
541 bool l = node_verify(tree, node, node->llink, black_node_count, leaf_black_node_count);
542 bool r = node_verify(tree, node, node->rlink, black_node_count, leaf_black_node_count);
543 return l && r;
544 }
545 return true;
546}

548bool
549rb_tree_verify(const rb_tree* tree)
550{
551 if (tree->root)
552 VERIFY(COLOR(tree->root) == RB_BLACK);
553 unsigned leaf_black_node_count = 0;
554 if (tree->root) {
555 VERIFY(tree->count > 0);
556 for (rb_node* node = tree->root; node; node = node->llink)
557 if (COLOR(node) == RB_BLACK)
558 leaf_black_node_count++;
559 } else {
560 VERIFY(tree->count == 0);
561 }
562 return node_verify(tree, NULL, tree->root, 0, leaf_black_node_count);
563}
