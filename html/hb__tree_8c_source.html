141size_t
142hb_tree_clear(hb_tree* tree, dict_delete_func delete_func)
143{
144 const size_t count = tree->count;
145
146 hb_node* node = tree->root;
147 while (node) {
148 if (node->llink || node->rlink) {
149 node = node->llink ? node->llink : node->rlink;
150 continue;
151 }
152
153 if (delete_func)
154 delete_func(node->key, node->datum);
155
156 hb_node* const parent = PARENT(node);
157 FREE(node);
158 *(parent ? (parent->llink == node ? &parent->llink : &parent->rlink) : &tree->root) = NULL;
159 node = parent;
160 }
161 ASSERT(tree->root == NULL);
162 tree->count = 0;
163 return count;
164}

166/* L: rotate |q| left */
167static bool
168rotate_l(hb_tree* restrict const tree, hb_node* restrict const q)
169{
170 hb_node* restrict const qp = PARENT(q);
171 hb_node* restrict const qr = q->rlink;
172
173 ASSERT(BAL_POS(q));
174
175 hb_node *restrict const qrl = qr->llink;
176 const int qr_bal = qr->bal & BAL_MASK;
177
178 /* q->parent <- qr; q->bal <- (qr_bal == 0); */
179 q->bal = (intptr_t)qr | (qr_bal == 0);
180 if ((q->rlink = qrl) != NULL)
181 qrl->bal = (intptr_t)q | (qrl->bal & BAL_MASK); /* qrl->parent <- q; */
182
183 /* qr->parent <- qp; qr->bal <- -(qr_bal == 0); */
184 qr->bal = (intptr_t)qp | ((qr_bal == 0) << 1);
185 qr->llink = q;
186 *(qp == NULL ? &tree->root : qp->llink == q ? &qp->llink : &qp->rlink) = qr;
187
188 return (qr_bal == 0);
189}

191/* R: rotate |q| right. */
192static bool
193rotate_r(hb_tree* restrict const tree, hb_node* restrict const q)
194{
195 hb_node* restrict const qp = PARENT(q);
196 hb_node* restrict const ql = q->llink;
197
198 ASSERT(BAL_NEG(q));
199
200 hb_node* restrict const qlr = ql->rlink;
201 const int ql_bal = ql->bal & BAL_MASK;
202
203 /* q->parent <- ql; q->bal <- -(ql_bal == 0); */
204 q->bal = (intptr_t)ql | ((ql_bal == 0) << 1);
205 if ((q->llink = qlr) != NULL)
206 qlr->bal = (intptr_t)q | (qlr->bal & BAL_MASK); /* qlr->parent <- q; */
207
208 /* ql->parent <- qp; ql->bal <- (ql_bal == 0); */
209 ql->bal = (intptr_t)qp | (ql_bal == 0);
210 ql->rlink = q;
211 *(qp == NULL ? &tree->root : qp->llink == q ? &qp->llink : &qp->rlink) = ql;
212
213 return (ql_bal == 0);
214}

216/* RL: Rotate |q->rlink| right, then |q| left. */
217static void
218rotate_rl(hb_tree* restrict const tree, hb_node* restrict const q)
219{
220 hb_node* restrict const qp = PARENT(q);
221 hb_node* restrict const qr = q->rlink;
222 hb_node* restrict const qrl = qr->llink;
223
224 ASSERT(BAL_POS(q));
225 ASSERT(BAL_NEG(qr));
226
227 const int qrl_bal = qrl->bal & BAL_MASK;
228 hb_node* restrict const qrll = qrl->llink;
229 hb_node* restrict const qrlr = qrl->rlink;
230
231 /* qrl->parent <- qp; qrl->bal <- 0; */
232 qrl->bal = (intptr_t)qp;
233 *(qp == NULL ? &tree->root : qp->llink == q ? &qp->llink : &qp->rlink) = qrl;
234 qrl->llink = q;
235 qrl->rlink = qr;
236
237 /* q->parent <- qrl; q->bal <- -(qrl_bal == 1); */
238 q->bal = (intptr_t)qrl | ((qrl_bal == 1) << 1);
239 if ((q->rlink = qrll) != NULL)
240 qrll->bal = (intptr_t)q | (qrll->bal & BAL_MASK);
241
242 /* qr->parent <- qrl; qr->bal <- (qrl_bal == -1); */
243 qr->bal = (intptr_t)qrl | (qrl_bal == 2);
244 if ((qr->llink = qrlr) != NULL)
245 qrlr->bal = (intptr_t)qr | (qrlr->bal & BAL_MASK);
246}

248/* LR: rotate |q->llink| left, then rotate |q| right. */
249static void
250rotate_lr(hb_tree* restrict const tree, hb_node* restrict const q)
251{
252 hb_node* restrict const qp = PARENT(q);
253 hb_node* restrict const ql = q->llink;
254 hb_node* restrict const qlr = ql->rlink;
255
256 ASSERT(BAL_NEG(q));
257 ASSERT(BAL_POS(ql));
258
259 const int qlr_bal = qlr->bal & BAL_MASK;
260 hb_node* restrict const qlrl = qlr->llink;
261 hb_node* restrict const qlrr = qlr->rlink;
262
263 /* qlr->parent <- qp; qlr->bal <- 0; */
264 qlr->bal = (intptr_t)qp;
265 *(qp == NULL ? &tree->root : qp->llink == q ? &qp->llink : &qp->rlink) = qlr;
266 qlr->llink = ql;
267 qlr->rlink = q;
268
269 /* q->parent <- qlr; q->bal = (qlr_bal == -1); */
270 q->bal = (intptr_t)qlr | (qlr_bal == 2);
271 if ((q->llink = qlrr) != NULL)
272 qlrr->bal = (intptr_t)q | (qlrr->bal & BAL_MASK);
273
274 /* ql->parent <- qlr; ql->bal <- -(qlr_bal == 1); */
275 ql->bal = (intptr_t)qlr | ((qlr_bal == 1) << 1);
276 if ((ql->rlink = qlrl) != NULL)
277 qlrl->bal = (intptr_t)ql | (qlrl->bal & BAL_MASK);
278}

280dict_insert_result
281hb_tree_insert(hb_tree* tree, void* key)
282{
283 int cmp = 0;
284 hb_node* node = tree->root;
285 hb_node* parent = NULL;
286 hb_node* q = NULL;
287 while (node) {
288 cmp = tree->cmp_func(key, node->key);
289 if (cmp < 0) {
290 parent = node; node = node->llink;
291 } else if (cmp > 0) {
292 parent = node; node = node->rlink;
293 } else
294 return (dict_insert_result) { &node->datum, false };
295 if (parent->bal & BAL_MASK)
296 q = parent;
297 }
298
299 hb_node* const add = node = node_new(key);
300 if (!node)
301 return (dict_insert_result) { NULL, false };
302
303 if (!(node->pptr = parent)) {
304 ASSERT(tree->count == 0);
305 ASSERT(tree->root == NULL);
306 tree->root = node;
307 } else {
308 if (cmp < 0)
309 parent->llink = node;
310 else
311 parent->rlink = node;
312
313 while (parent != q) {
314 ASSERT((parent->bal & BAL_MASK) == 0);
315 if (parent->llink == node)
316 parent->bal |= 2;
317 else
318 parent->bal |= 1;
319 node = parent;
320 parent = PARENT(parent);
321 }
322 if (q) {
323 ASSERT(q->bal & BAL_MASK);
324 if (q->llink == node) {
325 if (BAL_NEG(q)) { /* if q->balance == -1 */
326 if (BAL_POS(q->llink)) { /* if q->llink->balance == +1 */
327 tree->rotation_count += 2;
328 rotate_lr(tree, q);
329 } else {
330 tree->rotation_count += 1;
331 ASSERT(!rotate_r(tree, q));
332 }
333 } else { /* else, q->balance == +1 */
334 ASSERT(BAL_POS(q));
335 q->bal &= ~BAL_MASK; /* q->balance <- 0 */
336 }
337 } else {
338 ASSERT(q->rlink == node);
339 if (BAL_POS(q)) { /* if q->balance == +1 */
340 if (BAL_NEG(q->rlink)) { /* if q->rlink->balance == -1 */
341 tree->rotation_count += 2;
342 rotate_rl(tree, q);
343 } else {
344 tree->rotation_count += 1;
345 ASSERT(!rotate_l(tree, q));
346 }
347 } else { /* else, q->balance == -1 */
348 ASSERT(BAL_NEG(q));
349 q->bal &= ~BAL_MASK; /* q->balance <- 0 */
350 }
351 }
352 }
353 }
354 tree->count++;
355 return (dict_insert_result) { &add->datum, true };
356}

364static void
365remove_node(hb_tree* tree, hb_node* node)
366{
367 if (node->llink && node->rlink) {
368 hb_node* restrict out =
369 BAL_POS(node) ? tree_node_min(node->rlink) : tree_node_max(node->llink);
370 void* tmp;
371 SWAP(node->key, out->key, tmp);
372 SWAP(node->datum, out->datum, tmp);
373 node = out;
374 }
375
376 hb_node* p = PARENT(node);
377 hb_node* child = node->llink ? node->llink : node->rlink;
378 FREE(node);
379 tree->count--;
380 if (child)
381 child->bal = (intptr_t)p | (child->bal & BAL_MASK);
382 if (!p) {
383 tree->root = child;
384 return;
385 }
386
387 bool left = (p->llink == node);
388 if (left) {
389 p->llink = child;
390 } else {
391 ASSERT(p->rlink == node);
392 p->rlink = child;
393 }
394 node = child;
395
396 unsigned rotations = 0;
397 for (;;) {
398 if (left) {
399 ASSERT(p->llink == node);
400 if (BAL_POS(p)) { /* if p->balance == +1 */
401 if (BAL_NEG(p->rlink)) { /* if p->rlink->balance == -1 */
402 rotations += 2;
403 rotate_rl(tree, p);
404 } else {
405 rotations += 1;
406 if (rotate_l(tree, p))
407 break;
408 }
409 node = PARENT(p);
410 } else if (BAL_NEG(p)) { /* else if p->balance == -1 */
411 p->bal &= ~BAL_MASK; /* p->balance <- 0 */
412 node = p;
413 } else { /* else, p->balance == 0 */
414 ASSERT((p->bal & BAL_MASK) == 0);
415 p->bal |= 1; /* p->balance <- +1 */
416 break;
417 }
418 } else {
419 ASSERT(p->rlink == node);
420 if (BAL_NEG(p)) { /* if p->balance == -1 */
421 if (BAL_POS(p->llink)) { /* if p->llink->balance == +1 */
422 rotations += 2;
423 rotate_lr(tree, p);
424 } else {
425 rotations += 1;
426 if (rotate_r(tree, p))
427 break;
428 }
429 node = PARENT(p);
430 } else if (BAL_POS(p)) { /* else if p->balance == +1 */
431 p->bal &= ~BAL_MASK; /* p->balance <- 0 */
432 node = p;
433 } else { /* else, p->balance == 0 */
434 ASSERT((p->bal & BAL_MASK) == 0);
435 p->bal |= 2; /* p->balance <- -1 */
436 break;
437 }
438 }
439
440 if (!(p = PARENT(node)))
441 break;
442 if (p->llink == node) {
443 left = true;
444 } else {
445 ASSERT(p->rlink == node);
446 left = false;
447 }
448 }
449 tree->rotation_count += rotations;
450}

469bool
470hb_tree_select(hb_tree *tree, size_t n, const void **key, void **datum)
471{
472 if (n >= tree->count) {
473 if (key)
474 *key = NULL;
475 if (datum)
476 *datum = NULL;
477 return false;
478 }
479 hb_node* node;
480 if (n >= tree->count / 2) {
481 node = tree_node_max(tree->root);
482 n = tree->count - 1 - n;
483 while (n--)
484 node = node_prev(node);
485 } else {
486 node = tree_node_min(tree->root);
487 while (n--)
488 node = node_next(node);
489 }
490 if (key)
491 *key = node->key;
492 if (datum)
493 *datum = node->datum;
494 return true;
495}

543static bool
544node_verify(const hb_tree* tree, const hb_node* parent, const hb_node* node,
545 unsigned* height, size_t *count)
546{
547 if (!parent) {
548 VERIFY(tree->root == node);
549 } else {
550 if (parent->llink == node) {
551 if (node)
552 VERIFY(tree->cmp_func(parent->key, node->key) > 0);
553 } else {
554 ASSERT(parent->rlink == node);
555 if (node)
556 VERIFY(tree->cmp_func(parent->key, node->key) < 0);
557 }
558 }
559 if (node) {
560 int bal = node->bal & BAL_MASK;
561 VERIFY(bal >= 0);
562 VERIFY(bal <= 2);
563 if (bal == 2) {
564 VERIFY(node->llink != NULL);
565 bal = -1;
566 } else if (bal == 1) {
567 VERIFY(node->rlink != NULL);
568 }
569 VERIFY(PARENT(node) == parent);
570 unsigned lheight, rheight;
571 if (!node_verify(tree, node, node->llink, &lheight, count) ||
572 !node_verify(tree, node, node->rlink, &rheight, count))
573 return false;
574 VERIFY(bal == (int)rheight - (int)lheight);
575 if (height)
576 *height = MAX(lheight, rheight) + 1;
577 *count += 1;
578 } else {
579 if (height)
580 *height = 0;
581 }
582 return true;
583}

658bool
659hb_itor_remove(hb_itor* itor)
660{
661 if (!itor->node)
662 return false;
